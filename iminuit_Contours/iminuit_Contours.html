def pdf(x,a,b):
    x0 = 0.5 - b/3
    return x0 + a * x + b * x**2
def loglike(a,b):
    return -np.sum(np.log(pdf(df,a,b)**2))
min = iminuit.Minuit(loglike,a=0.5,b=0.5)
print(min.migrad())
plt.hist(df,bins=np.arange(-1,1.1,0.1),density=True)
xvals = np.linspace(-1,1,1000)
