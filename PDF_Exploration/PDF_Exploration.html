from scipy.stats import poisson
def nrepeats(N,size=5):
    mean,std = [],[]
    for i in range(N):
        events = poisson.rvs(2.0,size=size)
        mean.append(np.mean(events))
        std.append(np.std(events))
    return mean,std
def plotter(N,size=5):
    mean,std=nrepeats(N,size)
    plt.hist(mean,bins=99,histtype='step',label=f'Mu_5 mean = {np.mean(mean):.4f},Mu_5 variance = {np.var(mean):.4f}')
    plt.xlabel('Mean')
    plt.ylabel('Frequency')
    plt.title(f'Mean distribution for N={N}')
    plt.legend()
    plt.show()
    plt.hist(std,bins=99,histtype='step',label=f'Sigma_5 mean = {np.mean(std):.4f},Sigma_5 variance = {np.var(std):.4f}')
    plt.xlabel('Standard deviation')
    plt.ylabel('Frequency')
    plt.title(f'Std distribution for N={N}')
    plt.legend()
    plt.show()
plotter(10)
plotter(100)
plotter(1000)

from scipy.stats import gamma
means,vars = np.empty(1000),np.empty(1000)
for i in range(1000):
    gammadist = gamma.rvs(a=2,scale=1,size=100)
    means[i] = np.mean(gammadist)
    vars[i] = np.var(gammadist)
