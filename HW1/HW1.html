import matplotlib.pyplot as plt
from scipy.stats import norm
#loc is the mean, scale is the std
datapoints=norm.rvs(loc=3,scale=0.01,size=100000)
print(datapoints)
plt.hist(datapoints,bins=50,range=(2.95,3.05),density=True)
plt.title('Gaussian distribution histogram')
plt.xlabel('Value')
plt.ylabel('Occurrences')
plt.grid(True)

import numpy as np
xrange = np.linspace(2.95,3.05,1000)
#Normalized to 100k
pdfvalues=norm.pdf(xrange,loc=3,scale=0.01)*100000
plt.plot(xrange,pdfvalues)
plt.title('Gaussian PDF Normalized to 100k')
plt.xlabel('Value')
plt.ylabel('Probability Density')
plt.grid(True)
